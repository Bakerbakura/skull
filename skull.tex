\newcommand{\deflen}[2]{%      
  \expandafter\newlength\csname #1\endcsname
  \expandafter\setlength\csname #1\endcsname{#2}%
}

\deflen{cardW}{64mm}
\deflen{cardH}{64mm}

% assuming that emoji are square
\deflen{emojiW}{30mm}
\deflen{emojiH}{30mm}

\deflen{hSpace}{0mm}
\deflen{vSpace}{0mm}

\deflen{pageMarginV}{(\paperheight - \cardH * 4 - \vSpace * 3) * \real{0.5}}
\deflen{pageMarginH}{(\paperwidth - \cardW * 3 - \hSpace * 2) * \real{0.5}}

\deflen{emojiOffsetH}{\cardH * \real{0.5}}
\deflen{emojiOffsetW}{\cardW * \real{0.5}}

\deflen{cardMarginW}{4mm}
\deflen{cardMarginH}{4mm}
\deflen{borderThickness}{2mm}

\newcommand{\drawCard}{%
  \draw[color=black] (0,0) rectangle (\cardW, -\cardH);

  \draw[color=couleur\thestyle, line width=\borderThickness]
  (\cardMarginW,-\cardMarginH) rectangle (\cardW-\cardMarginW,\cardMarginH-\cardH);
  
  \node[inner sep=0pt] (emoji) at (\emojiOffsetW, -\emojiOffsetH) {%
    \ifthenelse{\therow < 3}{%
      \includegraphics[width=\emojiW]{rosette.pdf}
    }{%
      \includegraphics[width=\emojiW]{skull_crossbones.pdf}
    }
  };
}
