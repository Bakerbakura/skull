\newcommand{\setColours}{%
  \ifthenelse{\thepage=0}{%
    \ifthenelse{\thecol=0}{%
      \colorlet{currColour}{red}
    }{%
      \ifthenelse{\thecol=1}{%
        \colorlet{currColour}{green}
      }{%
        \colorlet{currColour}{blue}
      }
    }
  }{%
    \ifthenelse{\thecol=0}{%
      \colorlet{currColour}{cyan}
    }{%
      \ifthenelse{\thecol=1}{%
        \colorlet{currColour}{magenta}
      }{%
        \colorlet{currColour}{yellow}
      }
    }
  }
}

\newcommand{\drawCard}{%
  \draw[color=black] (0,0) rectangle (\cardW, -\cardH);

  \draw[color=currColour, line width=\borderThickness]
  (\cardMarginW,-\cardMarginH) rectangle (\cardW-\cardMarginW,\cardMarginH-\cardH);
  
  \node[inner sep=0pt] (emoji) at (\emojiOffsetW, -\emojiOffsetH) {%
    \ifthenelse{\therow < 3}{%
      \includegraphics[width=\emojiW]{rosette.svg}
    }{%
      \includegraphics[width=\emojiW]{skull_crossbones.svg}
    }
  };
}
