\newcommand{\drawCard}{%
  \draw[color=black] (0,0) rectangle (\cardW, -\cardH);

  \draw[color=couleur\thestyle, line width=\borderThickness]
  (\cardMarginW,-\cardMarginH) rectangle (\cardW-\cardMarginW,\cardMarginH-\cardH);
  
  \node[inner sep=0pt] (emoji) at (\emojiOffsetW, -\emojiOffsetH) {%
    \ifthenelse{\therow < 3}{%
      \includegraphics[width=\emojiW]{rosette.pdf}
    }{%
      \includegraphics[width=\emojiW]{skull_crossbones.pdf}
    }
  };
}
