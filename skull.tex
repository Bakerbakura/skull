\newcommand{\deflen}[2]{%      
  \expandafter\newlength\csname #1\endcsname
  \expandafter\setlength\csname #1\endcsname{#2}%
}

% number of players to print for (numCols*numSets)
\newcounter{numCols}
\setcounter{numCols}{4}
\newcounter{numSets}
\setcounter{numSets}{2}

% width and height of cards
\deflen{cardW}{64mm}
\deflen{cardH}{64mm}

% assuming that pics are square
\deflen{picW}{30mm}
\deflen{picH}{30mm}

% space between cards
\deflen{hSpace}{0mm}
\deflen{vSpace}{0mm}

% design of coloured borders of cards
\deflen{cardMarginW}{4mm}
\deflen{cardMarginH}{4mm}
\deflen{borderThickness}{2mm}

% colours for each player

% --- END OF CONFIGURATION --- %

\deflen{pageMarginV}{(\paperheight - \cardH*5 - \vSpace*4) * \real{0.5}}
\deflen{pageMarginH}{
  (\paperwidth - \cardW*\thenumCols - \hSpace*\thenumCols +\hSpace) * \real{0.5}
}

\deflen{picOffsetH}{\cardH * \real{0.5}}
\deflen{picOffsetW}{\cardW * \real{0.5}}

\newcounter{set}
\newcounter{row}
\newcounter{col}
\newcounter{style}

\newcommand{\drawCard}{%
  \draw[color=black] (0,0) rectangle (\cardW, -\cardH);

  \draw[color=couleur\thestyle, line width=\borderThickness]
  (\cardMarginW,-\cardMarginH) rectangle (\cardW-\cardMarginW,\cardMarginH-\cardH);
  
  \node[inner sep=0pt] (pic) at (\picOffsetW, -\picOffsetH) {%
    \ifthenelse{\therow < 3}{%
      \includegraphics[width=\picW]{rosette.pdf}
    }{%
      \ifthenelse{\therow = 3}{
        \includegraphics[width=\picW]{skull_crossbones.pdf}
      }{}
    }
  };
}

\pagenumbering{gobble}


\begin{document}

\forloop{set}{0}{\theset < \thenumSets}{
  \begin{tikzpicture}[overlay, shift=(current page.north west)]
    \forloop{col}{0}{\thecol < \thenumCols}{%
      \begin{scope}[shift={(\thecol*\cardW +\thecol*\hSpace +\pageMarginH, 0)}]
        \setcounter{style}{\thenumCols*\theset+\thecol}
        
        \forloop{row}{0}{\therow < 5}{%
          \begin{scope}[shift={(0, -\therow*\cardH -\therow*\vSpace -\pageMarginV)}]
            \drawCard
          \end{scope}
        }
      \end{scope}
    }
  \end{tikzpicture}
  \clearpage
}

\end{document}
